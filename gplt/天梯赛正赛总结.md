    cin >> n;

    {
        int a, b, c, d, e, f;

{
    sort(w.begin(), w.end(), [&](const Node &l, const Node &r){
        if(l.h1 == r.h1)
        {
            if(l.m1 == r.m1)    return l.s1 < r.s1;
            return l.m1 < r.m1;
        }
        return l.h1 < r.h1;
    });
